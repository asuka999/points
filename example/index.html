局限性

  • module.exports = var无法代替
  • require('./es6export').test并不是 引入default.test而是跟default同级的test
  • import a from './es5module.exports' 引入的是 module.exports 的东西

语法

export 支持语法

export default a
// exports.default = a
export default {  obj1 as o ,  foo1 as f}
// exports.default = { o : obj1 , f: foo1}

export {listen}
export {listen as listen}
// exports.listen = listen

export var e = 2.71828182846;
// exports.e = 2.71828182846

export function sum (){}
// exports.sum = function (){}

以下是错误示范

export { listen : listen}
export a
export function(){}

import 支持语法

import './file'
// require('./file')

import * as a from 'a'
// var a = require('a')

import a from from 'a'
// var a = require('a').default

import {a} from 'a'
// var a = reqiure('a').a

AMD

Asynchronous Module Definition(异步模块定义), 将等其所依赖的模块一一加载后才会被执行, 实现: RequireJS

define("module", ["./file1", "./file2"], function(d1, d2) {
  return exports;
});
require(["module", "../file"], function(module, file) { /* ... */ });

CMD

实现: seaJs

define(function(require, exports, module) {
  var a = require('./a')
  exports.a = 2.14
  module.exports = 2.14
});
